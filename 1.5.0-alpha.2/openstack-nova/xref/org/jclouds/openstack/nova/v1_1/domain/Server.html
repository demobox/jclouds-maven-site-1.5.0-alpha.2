62     /**
63      * Servers contain a status attribute that can be used as an indication of the current server
64      * state. Servers with an ACTIVE status are available for use.
65      * 
66      * Other possible values for the status attribute include: BUILD, REBUILD, SUSPENDED, RESIZE,
67      * VERIFY_RESIZE, REVERT_RESIZE, PASSWORD, REBOOT, HARD_REBOOT, DELETED, UNKNOWN, and ERROR.
68      * 
69      * @author Adrian Cole
70      */
71     public static enum Status {
72  
73        ACTIVE(NodeState.RUNNING), BUILD(NodeState.PENDING), REBUILD(NodeState.PENDING), SUSPENDED(NodeState.SUSPENDED), RESIZE(
74                 NodeState.PENDING), VERIFY_RESIZE(NodeState.PENDING), REVERT_RESIZE(NodeState.PENDING), PASSWORD(
75                 NodeState.PENDING), REBOOT(NodeState.PENDING), HARD_REBOOT(NodeState.PENDING), DELETED(
76                 NodeState.TERMINATED), UNKNOWN(NodeState.UNRECOGNIZED), ERROR(NodeState.ERROR), UNRECOGNIZED(
77                 NodeState.UNRECOGNIZED);
78  
79        private final NodeState nodeState;
80  
81        Status(NodeState nodeState) {
82           this.nodeState = nodeState;
83        }
84  
85        public String value() {
86           return name();
87        }
88  
89        public static Status fromValue(String v) {
90           try {
91              return valueOf(v.replaceAll("\\(.*", ""));
92           } catch (IllegalArgumentException e) {
93              return UNRECOGNIZED;
94           }
95        }
96  
97        public NodeState getNodeState() {
98           return nodeState;
99        }

510    /**
511     * @return the ip addresses assigned to the server
512     */
513    public Multimap<Type, Address> getAddresses() {
514 
515       Set<Address> privateAddresses = addresses.get(Address.Type.PRIVATE);
516 
517       if (privateAddresses != null && privateAddresses.size() > 1) {
518          return hackNeededForFloatingIpsFixedInEssex(privateAddresses);
519       } else {
520          return Multimaps2.fromOldSchool(addresses);
521       }
522 
523    }

525    private Multimap<Type, Address> hackNeededForFloatingIpsFixedInEssex(Set<Address> privateAddresses) {
526       Set<Address> publicAddresses = addresses.get(Address.Type.PUBLIC);
527       ImmutableSetMultimap.Builder<Type, Address> returnMapBuilder = new ImmutableSetMultimap.Builder<Type, Address>();
528       if (publicAddresses != null) {
529          returnMapBuilder.putAll(Address.Type.PUBLIC, publicAddresses);
530       }
531       returnMapBuilder.putAll(Address.Type.PRIVATE, Iterables.filter(privateAddresses, IsPrivateAddress.INSTANCE));
532       returnMapBuilder.putAll(Address.Type.PUBLIC, Iterables.filter(privateAddresses, Predicates
533                .not(IsPrivateAddress.INSTANCE)));
534       return returnMapBuilder.build();
535    }
